# Code Here
from sklearn.linear_model import LinearRegression
import pandas as pd
import numpy as np

# Drop rows with NaN
penguins_cleaned = penguins.dropna(subset=['bill_length_mm', 'bill_depth_mm'])

# Target and predictors labeled
y = penguins_cleaned['bill_depth_mm']
X = penguins_cleaned[['bill_length_mm']] # Use double brackets to keep X as a DataFrame

# Simple Linear Regression Model:
lr = LinearRegression()

# Set up training and test split
from sklearn.model_selection import train_test_split
X_train, X_test, y_train, y_test = train_test_split(X, y, test_size=0.25)

# Linear Regression model fit
lr_fit = lr.fit(X_train, y_train)

# Get predicted vals for the test data
y_pred_lr = lr_fit.predict(X_test)

# MSE for linear regression model
from sklearn.metrics import mean_squared_error
MSE_linear = mean_squared_error(y_test, y_pred_lr)


# Quadratic Model:
from sklearn.preprocessing import PolynomialFeatures
quad_model = PolynomialFeatures(degree=2)

# Make quadratic train and test
X_train_quad = quad_model.fit_transform(X_train)
X_test_quad = quad_model.transform(X_test)

# Fit the model with quadratic features
lr2 = LinearRegression()
lr2_fit = lr2.fit(X_train_quad, y_train)

# Predict on the test set for quad
y_pred_quad = lr2_fit.predict(X_test_quad)

# MSE for quadratic model
MSE_quad = mean_squared_error(y_test, y_pred_quad)


# Cubic Model:
cubic_model = PolynomialFeatures(degree=3)

# Make cubic train and test
X_train_cubic = cubic_model.fit_transform(X_train)
X_test_cubic = cubic_model.transform(X_test)

# Fit the model with cubic features
lr3 = LinearRegression()
lr3_fit = lr3.fit(X_train_cubic, y_train)

# Predict on the test set for cubic
y_pred_cubic = lr3_fit.predict(X_test_cubic)

# MSE for cubic model
MSE_cubic = mean_squared_error(y_test, y_pred_cubic)


# 10 Polynomial Model:
tenpoly_model = PolynomialFeatures(degree=10)

# Make 10 polynomial train and test
X_train_tenpoly = tenpoly_model.fit_transform(X_train)
X_test_tenpoly = tenpoly_model.transform(X_test)

# Fit the model with 10 polynomial features
lr10 = LinearRegression()
lr10_fit = lr10.fit(X_train_tenpoly, y_train)

# Predict on the test test for 10 poly
y_pred_tenpoly = lr10_fit.predict(X_test_tenpoly)

# MSE for 10 polynomial model
MSE_tenpoly = mean_squared_error(y_test, y_pred_tenpoly)
